<script>
  const audio = document.getElementById("bgAudio");

  // Pastikan audio tetap bermain meskipun iframe di atas
  window.addEventListener('load', () => {
    audio.play().catch((e) => {
      console.warn('Audio tidak bisa autoplay karena izin browser.');
    });
  });

  // Jika pengunjung keluar dari halaman, stop audio
  window.addEventListener('beforeunload', () => {
    audio.pause();
    audio.currentTime = 0;
  });
</script>
